tidyfaf

tidyfaf - Tidy access to FAF freight flow data.

Inspired by tidycensus, provides chainable query builders for FAF regional flows, state flows, network data, and forecasts.

Examples

>>> import tidyfaf as faf

>>> # Discover available data
>>> faf.available_commodities(search='electronics')
>>> faf.available_zones(search='california')

>>> # Query regional flows
>>> query = (faf.FAFQuery()
...     .origin_states(['California', 'Texas'])
...     .destination_zones([111])
...     .commodities(['Electronics'])
...     .years([2020, 2030])
... )
>>> df = query.get()

>>> # Network analysis
>>> network = faf.NetworkQuery().routes(['I-5']).states(['CA', 'OR']).get()

 69    def _get_pushdown_filters(self):
 70        """Convert high-level filters to PyArrow DNF filters."""
 71        filters = []
 72        
 73        # Helper to append simple IN filter
 74        def add_in_filter(col, values):
 75            if values:
 76                # PyArrow filter format: [(col, 'in', values), ...]
 77                # Note: read_parquet filters argument expects a "DNF" list of lists.
 78                # [[(col, op, val)], [(col, op, val)]] means OR
 79                # [[(col, op, val), (col, op, val)]] means AND
 80                # We want AND between different fields.
 81                # So we build a single list of tuples for AND?
 82                # Wait, fastparquet/pyarrow 'filters' arg:
 83                # List of tuples: [('col', '==', val), ...] -> AND
 84                # List of lists of tuples: [[('col', '==', val)], ...] -> OR (Wait, check docs)
 85                # PyArrow dataset: DNF. List of List of Tuples. Outer list is OR. Inner list is AND.
 86                # Pandas read_parquet "filters": "List of tuples or list of lists of tuples"
 87                # "List of tuples" is AND-ed.
 88                filters.append((col, 'in', values))
 89
 90        # Origin Zones (Direct)
 91        if 'origin_zones' in self._filters:
 92            add_in_filter('dms_orig', self._filters['origin_zones'])
 93
 94        # Origin States (Convert to Zones)
 95        elif 'origin_states' in self._filters:
 96            # We need to find all zones for these states
 97            state_codes = self._filters['origin_states']
 98            # Filter metadata zones
 99            zones_df = self._metadata.zones
100            
101            valid_zones = []
102            for state in state_codes:
103                # Column is 'Numeric Label' in metadata sheet
104                state_zones = zones_df[zones_df['Numeric Label'] // 10 == state]['Numeric Label'].tolist()
105                valid_zones.extend(state_zones)
106            
107            if valid_zones:
108                add_in_filter('dms_orig', valid_zones)
109
110        # Destination Zones (Direct)
111        if 'destination_zones' in self._filters:
112            add_in_filter('dms_dest', self._filters['destination_zones'])
113            
114        # Destination States (Convert to Zones)
115        elif 'destination_states' in self._filters:
116            state_codes = self._filters['destination_states']
117            zones_df = self._metadata.zones
118            valid_zones = []
119            for state in state_codes:
120                state_zones = zones_df[zones_df['Numeric Label'] // 10 == state]['Numeric Label'].tolist()
121                valid_zones.extend(state_zones)
122            
123            if valid_zones:
124                add_in_filter('dms_dest', valid_zones)
125
126        # Commodities
127        if 'commodities' in self._filters:
128            add_in_filter('sctg2', self._filters['commodities'])
129
130        # Modes
131        if 'modes' in self._filters:
132            add_in_filter('dms_mode', self._filters['modes'])
133
134        # Trade Types
135        if 'trade_types' in self._filters:
136            add_in_filter('trade_type', self._filters['trade_types'])
137
138        # Return single inner list for AND behavior across all conditions
139        return [filters] if filters else None

392    def _apply_filters(self, df):
393        """Apply all filters to DataFrame."""
394
395        # Origin filtering (handle state vs zone)
396        if 'origin_states' in self._filters:
397            # Filter by state: dms_orig // 10 == state_fips
398            # FAF zone codes are structured as: state_fips * 10 + zone_within_state
399            state_codes = self._filters['origin_states']
400            df = df[(df['dms_orig'] // 10).isin(state_codes)]
401
402        if 'origin_zones' in self._filters:
403            df = df[df['dms_orig'].isin(self._filters['origin_zones'])]
404
405        # Destination filtering
406        if 'destination_states' in self._filters:
407            state_codes = self._filters['destination_states']
408            df = df[(df['dms_dest'] // 10).isin(state_codes)]
409
410        if 'destination_zones' in self._filters:
411            df = df[df['dms_dest'].isin(self._filters['destination_zones'])]
412
413        # Commodity filtering
414        if 'commodities' in self._filters:
415            df = df[df['sctg2'].isin(self._filters['commodities'])]
416
417        # Mode filtering
418        if 'modes' in self._filters:
419            df = df[df['dms_mode'].isin(self._filters['modes'])]
420
421        # Trade type filtering
422        if 'trade_types' in self._filters:
423            df = df[df['trade_type'].isin(self._filters['trade_types'])]
424
425        # Threshold filtering
426        if 'min_tons' in self._filters:
427            value, year = self._filters['min_tons']
428            col = f'tons_{year}'
429            if col in df.columns:
430                df = df[df[col] >= value]
431
432        if 'min_value' in self._filters:
433            value, year = self._filters['min_value']
434            col = f'value_{year}'
435            if col in df.columns:
436                df = df[df[col] >= value]
437
438        return df

442    def group_by(self, fields, metrics=None, years=None):
443        """
444        Group by specified fields and aggregate metrics.
445
446        Parameters
447        ----------
448        fields : list of str or str
449            Fields to group by (e.g., ['dms_orig', 'sctg2'] or 'dms_orig')
450        metrics : list of str, default ['tons', 'value']
451            Metrics to aggregate
452        years : list of int, optional
453            Years to aggregate. If None, uses years from filter.
454
455        Returns
456        -------
457        pd.DataFrame
458            Aggregated data
459
460        Examples
461        --------
462        >>> query.group_by(['dms_orig', 'sctg2'], metrics=['tons'], years=[2020])
463        """
464        if isinstance(fields, str):
465            fields = [fields]
466
467        if metrics is None:
468            metrics = ['tons', 'value']
469
470        df = self.get()
471
472        # Determine years
473        if years is None:
474            years = self._filters.get('years', [2020])
475
476        # Build aggregation dict
477        agg_dict = {}
478        for metric in metrics:
479            for year in years:
480                col = f'{metric}_{year}'
481                if col in df.columns:
482                    agg_dict[col] = 'sum'
483
484        if not agg_dict:
485            return pd.DataFrame()  # No columns to aggregate
486
487        return df.groupby(fields).agg(agg_dict).reset_index()

124    def _reshape_scenarios(self, df):
125        """
126        Reshape wide format with base/high/low columns to long format.
127        """
128        print(f"DEBUG: _reshape_scenarios input df shape: {df.shape}")
129        print(f"DEBUG: _reshape_scenarios input df columns: {df.columns.tolist()}")
130
131        # Identify metadata columns
132        meta_cols = ['dms_orig', 'dms_dest', 'sctg2', 'dms_mode',
133                    'trade_type', 'dist_band', 'fr_orig', 'fr_dest',
134                    'fr_inmode', 'fr_outmode']
135        meta_cols = [c for c in meta_cols if c in df.columns]
136
137        # Get forecast years from filter
138        years = self._filters.get('years', [2030, 2035, 2040, 2045, 2050])
139        # Filter to forecast years only (>= 2030)
140        years = [y for y in years if y >= 2030]
141
142        scenarios_requested = self._filters.get('scenarios', ['base', 'high', 'low'])
143
144        dfs = []
145
146        for scenario in scenarios_requested:
147            scenario_df = df[meta_cols].copy()
148
149            for year in years:
150                # Base scenario uses tons_YEAR, value_YEAR
151                # High/Low scenarios use tons_YEAR_high, tons_YEAR_low, etc.
152                if scenario == 'base':
153                    tons_col = f'tons_{year}'
154                    value_col = f'value_{year}'
155                else:
156                    tons_col = f'tons_{year}_{scenario}'
157                    value_col = f'value_{year}_{scenario}'
158
159                # Add columns if they exist
160                if tons_col in df.columns:
161                    scenario_df[f'tons_{year}'] = df[tons_col]
162                if value_col in df.columns:
163                    scenario_df[f'value_{year}'] = df[value_col]
164
165            scenario_df['scenario'] = scenario
166            dfs.append(scenario_df)
167
168        # Concatenate all scenarios
169        result = pd.concat(dfs, ignore_index=True)
170        print(f"DEBUG: _reshape_scenarios concat result shape: {result.shape}")
171
172        # Now convert to long format with year dimension
173        metric_cols = [col for col in result.columns
174                      if col.startswith('tons_') or col.startswith('value_')]
175        
176        print(f"DEBUG: _reshape_scenarios metric_cols: {metric_cols}")
177        id_cols = meta_cols + ['scenario']
178
179        result_long = result.melt(
180            id_vars=id_cols,
181            value_vars=metric_cols,
182            var_name='metric_year',
183            value_name='value'
184        )
185
186        # Split metric_year into metric and year
187        result_long['metric'] = result_long['metric_year'].str.split('_').str[0]
188        result_long['year'] = result_long['metric_year'].str.split('_').str[1].astype(int)
189        result_long = result_long.drop(columns=['metric_year'])
190
191        # Fill NaNs in index columns to prevent pivot_table from dropping rows
192        result_long[id_cols] = result_long[id_cols].fillna(-1)
193
194        # Pivot so each metric is a column
195        result_long = result_long.pivot_table(
196            index=id_cols + ['year'],
197            columns='metric',
198            values='value'
199        ).reset_index()

202        # Note: -1 is safe for FAF codes which are positive integers
203        import numpy as np
